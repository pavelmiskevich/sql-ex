Найдите среднюю скорость ПК-блокнотов, цена которых превышает 1000 дол.
Получение итоговых значений
-- Процесс -------------------------------------------------------------

-- Решение -------------------------------------------------------------
select avg(speed) from Laptop where price > 1000

-- GIT HUB
SELECT AVG(speed) 
    FROM laptop 
    WHERE price > 1000
